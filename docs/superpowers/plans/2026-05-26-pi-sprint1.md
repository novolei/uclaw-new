### Step 1.3：实现 RollingTailBuffer

- [ ] 在 `shell.rs` 的 `use` 块之后，`BashTool` struct 定义之前，插入：

  ```rust
  use std::collections::VecDeque;
  use uclaw_utils_home::uclaw_home_pathbuf;

  /// 保留最近 N 字节的滚动尾部缓冲区。
  /// 超出 capacity 时，最旧的字节从头部丢弃。
  struct RollingTailBuffer {
      capacity: usize,
      buf: VecDeque<u8>,
      total_written: usize,
      dropped: usize,
  }

  impl RollingTailBuffer {
      fn new(capacity: usize) -> Self {
          Self {
              capacity,
              buf: VecDeque::with_capacity(capacity),
              total_written: 0,
              dropped: 0,
          }
      }

      fn push_bytes(&mut self, data: &[u8]) {
          self.total_written += data.len();
          for &byte in data {
              if self.buf.len() >= self.capacity {
                  self.buf.pop_front();
                  self.dropped += 1;
              }
              self.buf.push_back(byte);
          }
      }

      /// 生成返回给 LLM 的字符串。
      /// `temp_path` 有值时在截断头注中附加文件路径。
      fn to_truncated_string(&self, temp_path: Option<&std::path::Path>) -> String {
          let (a, b) = self.buf.as_slices();
          let content = String::from_utf8_lossy(a).to_string()
              + &String::from_utf8_lossy(b);
          if self.dropped > 0 {
              let path_note = temp_path
                  .map(|p| format!("，完整输出已保存至 {}", p.display()))
                  .unwrap_or_default();
              format!(
                  "[输出已截断：共 {} 字节，显示最后 {} 字节{path_note}]\n\n{}",
                  self.total_written,
                  self.buf.len(),
                  content
              )
          } else {
              content
          }
      }
  }
  ```

### Step 2.3：实现 SessionFileOps

- [ ] 在 `file_ops.rs` 的测试模块之前添加实现：

  ```rust
  /// 会话文件操作记录（跨压缩周期累积）。
  /// Pi 对应：`SessionFileOps` in `compaction/utils.ts`。
  #[derive(Debug, Clone, Default, Serialize, Deserialize)]
  pub struct SessionFileOps {
      /// 已读取的文件路径集合。
      pub read: HashSet<PathBuf>,
      /// 已写入（write_file / write）的文件路径集合。
      pub written: HashSet<PathBuf>,
      /// 已编辑（edit / apply_patch / str_replace）的文件路径集合。
      pub edited: HashSet<PathBuf>,
  }

  impl SessionFileOps {
      /// 从工具调用中提取并记录文件操作。
      /// `tool_name` 必须是工具的规范名称（如 `"read_file"`, `"edit"`）。
      pub fn track_tool_call(&mut self, tool_name: &str, args: &serde_json::Value) {
          let path = match args
              .get("path")
              .or_else(|| args.get("file_path"))
              .and_then(|p| p.as_str())
          {
              Some(p) if !p.is_empty() => PathBuf::from(p),
              _ => return,
          };

          match tool_name {
              "read_file" | "read" => { self.read.insert(path); }
              "write_file" | "write" => { self.written.insert(path); }
              "edit" | "apply_patch" | "str_replace" | "str_replace_editor" => {
                  self.edited.insert(path);
              }
              _ => {}
          }
      }

      /// 合并另一个 FileOps（用于跨压缩周期累积）。
      pub fn merge(&mut self, other: &SessionFileOps) {
          self.read.extend(other.read.iter().cloned());
          self.written.extend(other.written.iter().cloned());
          self.edited.extend(other.edited.iter().cloned());
      }

      /// 生成摘要尾注，附加到压缩输出末尾。
      /// 空操作时返回空字符串（不在摘要中产生占位符）。
      pub fn format_for_summary(&self) -> String {
          let mut parts: Vec<String> = Vec::new();

          if !self.read.is_empty() {
              let mut files: Vec<String> = self
                  .read
                  .iter()
                  .map(|p| p.display().to_string())
                  .collect();
              files.sort();
              parts.push(format!("**已读取文件**: {}", files.join(", ")));
          }

          let modified: HashSet<&PathBuf> = self.written.union(&self.edited).collect();
          if !modified.is_empty() {
              let mut files: Vec<String> = modified
                  .iter()
                  .map(|p| p.display().to_string())
                  .collect();
              files.sort();
              parts.push(format!("**已修改文件**: {}", files.join(", ")));
          }

          if parts.is_empty() {
              String::new()
          } else {
              format!(
                  "\n\n---\n\n## 文件操作记录（跨压缩持久）\n\n{}",
                  parts.join("\n")
              )
          }
      }
  }
  ```

### Step 3.3：实现 format_for_system_prompt_xml

- [ ] 在 `skills.rs` 的 `impl SkillsRegistry` 块中，在 `combined_system_prompt()` 方法之后添加：

  ```rust
  /// 以 Pi 规范的 XML 格式生成 Skills 系统 prompt 块。
  ///
  /// 精确还原 Pi `system-prompt.ts::formatSkillsForSystemPrompt()` 的输出：
  ///   - 3 条固定指令头行（告知模型如何使用 skills）
  ///   - `<available_skills>` XML（每个 skill 四行缩进）
  ///
  /// 返回空字符串表示没有任何可见 skill。
  pub fn format_for_system_prompt_xml(&self) -> String {
      let visible: Vec<(&SkillManifest, &LoadedSkill)> = self
          .list_enabled()
          .into_iter()
          .filter_map(|m| self.skills.get(&m.name).map(|loaded| (m, loaded)))
          .collect();

      if visible.is_empty() {
          return String::new();
      }

      let mut lines = vec![
          "The following skills provide specialized instructions for specific tasks.",
          "Read the full skill file when the task matches its description.",
          "When a skill file references a relative path, resolve it against the skill directory \
           (parent of SKILL.md / dirname of the path) and use that absolute path in tool commands.",
          "",
          "<available_skills>",
      ];
      let skill_lines: Vec<String> = visible.iter().flat_map(|(manifest, loaded)| {
          let desc = manifest
              .description
              .as_deref()
              .unwrap_or("")
              .to_string();
          let location = loaded.path.to_string_lossy().to_string();
          vec![
              "  <skill>".to_string(),
              format!("    <name>{}</name>", Self::escape_xml(&manifest.name)),
              format!("    <description>{}</description>", Self::escape_xml(&desc)),
              format!("    <location>{}</location>", Self::escape_xml(&location)),
              "  </skill>".to_string(),
          ]
      }).collect();

      let mut result = lines.join("\n");
      if !skill_lines.is_empty() {
          result.push('\n');
          result.push_str(&skill_lines.join("\n"));
      }
      result.push_str("\n</available_skills>");
      result
  }

  fn escape_xml(s: &str) -> String {
      s.replace('&', "&amp;")
       .replace('<', "&lt;")
       .replace('>', "&gt;")
       .replace('"', "&quot;")
       .replace('\'', "&apos;")
  }
  ```

  注：如果 `LoadedSkill` 没有 `path: PathBuf` 字段，查看 `skills.rs` 第 240 行附近的 `LoadedSkill` 定义，使用实际字段名（可能是 `file_path` 或从 manifest 中取 slug）。

### Step 4.4：在 execute_tool_calls() 中添加并行分派路径

`execute_tool_calls()` 的大部分逻辑（安全检查、approval 流程）必须保持串行。只有通过所有检查后的实际执行阶段，才能并行化。

方案：先串行完成所有预检和 approval，收集"已批准待执行的工具列表"，然后分批执行：

- [ ] 在 `execute_tool_calls()` 的 `for tc in &tool_calls` 循环结束位置（约第 2700 行之后），找到当前单个工具执行模式：

  ```rust
  // 当前模式（约第 2641–2668 行）：
  let execute_result = match tokio::task::spawn(async move { ... }).await { ... };
  ```

  **不修改这段代码**。在 for 循环外层添加一个"并行批次"逻辑：

- [ ] 在 `execute_tool_calls()` 函数签名之后、`for tc in &tool_calls` 循环**之前**，添加并行批次变量：

  ```rust
  // 收集可并行执行的工具调用（已通过预检）
  // key = tool_call_id, value = (tool_name, args)
  let mut parallel_batch: Vec<(String, String, serde_json::Value)> = Vec::new(); // (id, name, args)
  ```

  然后在 for 循环内，当工具通过 approval 且 `is_parallel_safe(&tc.name)` 时，将其推入 `parallel_batch` 而不是立即执行，并跳过原有的 `tokio::task::spawn` 调用：

  ```rust
  // 在 approval 检查通过后，替换原来的立即执行：
  if is_parallel_safe(&tc.name) {
      parallel_batch.push((tc.id.clone(), tc.name.clone(), tool_args_for_spawn.clone()));
      // 跳过后续 execute_result 处理，在循环外统一执行
      continue;
  }
  // 非并行安全工具：走原有 tokio::task::spawn 路径（代码不变）
  ```

- [ ] 在 `for tc` 循环**结束之后**，执行并行批次：

  ```rust
  // 执行并行安全工具批次
  if !parallel_batch.is_empty() {
      use tokio::task::JoinSet;
      let mut join_set: JoinSet<(String, String, Result<crate::agent::tools::tool::ToolOutput, crate::agent::tools::tool::ToolError>)> = JoinSet::new();

      for (id, name, args) in parallel_batch {
          let tools_arc = Arc::clone(&self.tools);
          let ctx = ToolExecutionContext::agent_turn(
              self.conversation_id.clone(),
              id.clone(),
              self.workspace_root.clone(),
              self.safety_mode.clone(),
          );
          join_set.spawn(async move {
              let result = match tools_arc.get(&name) {
                  Some(t) => execute_tool_with_context(t, args, &ctx).await,
                  None => Err(crate::agent::tools::tool::ToolError::NotFound(name.clone())),
              };
              (id, name, result)
          });
      }

      // 收集结果（顺序不定，用 id 关联原始 ToolCall）
      while let Some(join_result) = join_set.join_next().await {
          match join_result {
              Ok((id, name, Ok(output))) => {
                  self.emit_tool_result(&name, &id, &output);
                  reason_ctx.file_ops.track_tool_call(&name, &output.result);
                  reason_ctx.messages.push(ChatMessage::user_tool_result(
                      &id,
                      &output.content_text(),
                      false,
                  ));
              }
              Ok((id, name, Err(e))) => {
                  tracing::warn!(tool = %name, error = %e, "parallel tool failed");
                  let err_output = crate::agent::tools::tool::ToolOutput::error(&e.to_string(), 0);
                  self.emit_tool_result(&name, &id, &err_output);
                  reason_ctx.messages.push(ChatMessage::user_tool_result(
                      &id, &e.to_string(), true,
                  ));
              }
              Err(join_err) => {
                  tracing::error!(error = %join_err, "parallel tool join error");
              }
          }
      }
  }
  ```

  注：`content_text()` 可能需要确认 `ToolOutput` 上的实际方法名（`grep -n "fn content_text\|fn output_text\|fn text" src-tauri/src/agent/tools/tool.rs | head -5`）。
